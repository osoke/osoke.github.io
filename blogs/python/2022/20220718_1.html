    python 读取不同类型剪贴板数据

    osoke 2022/7/18 读取剪贴板ctypes

    在 Windows 中, 我们都知道: 你可以在文件管理器中 ctrl+c 复制一些文件, 然后切换到别的文件夹中 ctrl+v 粘贴, 文件就被轻松的转移过来了。但是, 我们自己开发的应用, 默认情况下是没有办法这样操作的。不过, 我们可以手动添加这样的功能, 增加我们开发的应用与操作系统之间的互动性

    接下来主要介绍如何用 python 读取 Windows 剪贴板数据, 需要用到 ctypes 这个库

    # ctypes 简介

    ctypes 是 Python 的外部函数库, 使用它可以方便的调用 c 语言编写的动态链接库函数(dll)

    # 读取复制的 text 文本

    使用 ctypes, 为了避免类型错误, 一般需要显式声明参数与返回值类型

    import ctypes  
import ctypes.wintypes as w  
  
# 剪贴板数据类型  
CF_UNICODETEXT = 13  # unicode 字符  

# ------------------------  
u32 = ctypes.windll.user32    # 加载 user32   库  
k32 = ctypes.windll.kernel32  # 加载 kernel32 库  
s32 = ctypes.windll.shell32   # 加载 shell32  库  
  
# ------------------------  
OpenClipboard = u32.OpenClipboard  # func ▶ open 剪贴板  
OpenClipboard.argtypes = w.HWND,   # args (HWND, )  
OpenClipboard.restype = w.BOOL     # 返回 -> BOOL  

GetClipboardData = u32.GetClipboardData  # func ▶ 获取剪贴板数据  
GetClipboardData.argtypes = w.UINT,      # args (UINT, )  
GetClipboardData.restype = w.HANDLE      # 返回 -> HANDLE  
# ------------------------  
GlobalLock = k32.GlobalLock        # func ▶ 全局内存锁定  
GlobalLock.argtypes = w.HGLOBAL,   # args (HGLOBAL, )  
GlobalLock.restype = w.LPVOID      # 返回 -> LPVOID  

GlobalUnlock = k32.GlobalUnlock    # func ▶ 全局内存解锁  
GlobalUnlock.argtypes = w.HGLOBAL, # args (HGLOBAL, )  
GlobalUnlock.restype = w.BOOL      # 返回 -> BOOL  
# ------------------------  
CloseClipboard = u32.CloseClipboard  # func ▶ close 剪贴板  
CloseClipboard.argtypes = None       # args 无  
CloseClipboard.restype = w.BOOL      # 返回 -> BOOL 

def get_clipboard_text():  
    """获取剪贴板-文本"""  
    text = ""  
    if OpenClipboard(None):  
        h_clip_mem = GetClipboardData(CF_UNICODETEXT)  
        text = ctypes.wstring_at(GlobalLock(h_clip_mem))  # 读取内存地址 -> 宽字符 str        
        
        GlobalUnlock(h_clip_mem)  
        CloseClipboard()  
    return text  

if __name__ == '__main__':  
    print(get_clipboard_text())

    在以上代码基础上, 继续编写代码

    首先指定剪贴板数据类型

    CF_HDROP = 15        # 文件句柄 ▶ 指向 HDROP 这种结构, 通过 DragQueryFile 函数来获得文件信息

DragQueryFile.argtypes = [w.HANDLE, w.UINT, ctypes.c_void_p, w.UINT]

    读取剪贴板文件列表

    def get_clipboard_files():  
    """获取剪贴板-文件列表"""  
    files = []  
    if OpenClipboard(None):  
        h_hdrop = GetClipboardData(CF_HDROP)  # 获取剪贴板句柄  
        if h_hdrop:  
            GlobalLock(h_hdrop)  
  
            file_count = DragQueryFile(h_hdrop, -1, None, 0)  # 返回剪贴板文件数量  
            for i in range(file_count):  
                # 创建 buf 
                buf = ctypes.create_unicode_buffer(260)  
                # 从'全局内存对象'中提取每个文件的路径 ▶ 存入 buf  
                DragQueryFile(h_hdrop, i, buf, ctypes.sizeof(buf)) 
                files.append(buf.value)  
  
            GlobalUnlock(h_hdrop)  
            CloseClipboard()  
    return files

    # 在此基础上增加拖拽功能

    之前写过一篇文章 Tkinter 借助 windnd 实现拖放文件 (opens new window), 结合本文的读取剪贴板操作, 可以进一步提高桌面应用的易用性; 所以可以将它们整合到一起, 不过在整合过程中, 遇到一点小问题, 在这里记录一下

    由于「windnd 拖放操作」和「读取剪贴板」内部都是调用的 ctypes 库, 但是调用 DragQueryFileW 函数的方式不一样, 参数类型不一致, 这就需要调整 DragQueryFileW 函数的参数, 与 windnd 保持一致, 否则会导致类型错误

    修改上述代码:

    加载 DragQueryFileW 函数

    • 调整1: 去除 args 类型设置
    • 调整2: 按照 windnd 的逻辑, 获取 handle 数据类型
    import platform  

DragQueryFile = s32.DragQueryFileW 
# DragQueryFile.argtypes = [w.HANDLE, w.UINT, ctypes.c_void_p, w.UINT] 

# argtypes 与 windnd 保持一致  
if platform.architecture()[0] == "32bit":  
    handle_type = ctypes.c_ulong  
elif platform.architecture()[0] == "64bit":  
    handle_type = ctypes.c_uint64

    import ctypes
import ctypes.wintypes as w
import platform


CF_UNICODETEXT = 13  
CF_HDROP = 15        

# ------------------------
u32 = ctypes.windll.user32    
k32 = ctypes.windll.kernel32  
s32 = ctypes.windll.shell32   

# ------------------------
OpenClipboard = u32.OpenClipboard  
OpenClipboard.argtypes = w.HWND,   
OpenClipboard.restype = w.BOOL     

GetClipboardData = u32.GetClipboardData  
GetClipboardData.argtypes = w.UINT,      
GetClipboardData.restype = w.HANDLE      

# ------------------------
GlobalLock = k32.GlobalLock        
GlobalLock.argtypes = w.HGLOBAL,   
GlobalLock.restype = w.LPVOID      

GlobalUnlock = k32.GlobalUnlock    
GlobalUnlock.argtypes = w.HGLOBAL, 
GlobalUnlock.restype = w.BOOL      

# ------------------------
CloseClipboard = u32.CloseClipboard  
CloseClipboard.argtypes = None       
CloseClipboard.restype = w.BOOL      

# ------------------------
DragQueryFile = s32.DragQueryFileW 

if platform.architecture()[0] == "32bit":
    handle_type = ctypes.c_ulong
elif platform.architecture()[0] == "64bit":
    handle_type = ctypes.c_uint64


def get_clipboard_text():
    """获取剪贴板-文本"""
    text = ""
    if OpenClipboard(None):
        h_clip_mem = GetClipboardData(CF_UNICODETEXT)
        text = ctypes.wstring_at(GlobalLock(h_clip_mem))  
        GlobalUnlock(h_clip_mem)
        CloseClipboard()
    return text

def get_clipboard_files():
    """获取剪贴板-文件列表"""
    files = []
    if OpenClipboard(None):
        h_hdrop = GetClipboardData(CF_HDROP)  
        if h_hdrop:
            GlobalLock(h_hdrop)

            file_count = DragQueryFile(handle_type(h_hdrop), -1, None, 0)  
            for i in range(file_count):
                buf = ctypes.create_unicode_buffer(260)
                DragQueryFile(handle_type(h_hdrop), i, buf, ctypes.sizeof(buf)) 
                files.append(buf.value)

            GlobalUnlock(h_hdrop)
            CloseClipboard()
    return files

if __name__ == '__main__':
    print(get_clipboard_text())
    # print(get_clipboard_files())
